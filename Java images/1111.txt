    private void button1ActionPerformed(java.awt.event.ActionEvent evt) {                                        
        // TODO add your handling code here:
        if(choice1.getSelectedIndex() == 0)
        {
            this.setVisible(false);
            new VegetablesAndFriuts().setVisible(true);
        }
        if(choice1.getSelectedIndex()==1)
       {
           this.setVisible(false);
           new GroceryAndStaples().setVisible(true);
       }
       if(choice1.getSelectedIndex()==2)
       {
           this.setVisible(false);
           new DairyProducts().setVisible(true);
       }
       if(choice1.getSelectedIndex()==3)
       {
           this.setVisible(false);
           new SaucesAndKetchup().setVisible(true);
       }
       if(choice1.getSelectedIndex()==4)
       {
           this.setVisible(false);
           new PastasAndNoodles().setVisible(true);
       }
    }
